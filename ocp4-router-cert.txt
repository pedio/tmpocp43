## Obtain the router cert and key

# get router cert CA
oc get secrets/router-ca -n openshift-ingress-operator -o template='{{index .data "tls.crt"}}' | base64 -d > ocprouterca.crt

# get router cert
oc get secrets/router-certs-default -n openshift-ingress -o template='{{index .data "tls.crt"}}' | base64 -d > ocprouter.crt

# get router key
oc get secrets/router-certs-default -n openshift-ingress -o template='{{index .data "tls.key"}}' | base64 -d > ocprouter.key

# verify the router cert
openssl x509 -in ocprouter.crt -noout -text
